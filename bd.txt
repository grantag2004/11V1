CREATE TABLE goods(
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	name TEXT NOT NULL UNIQUE,
	amount INTEGER NOT NULL,
	price INTEGER NOT NULL,
	info TEXT NOT NULL UNIQUE, 
	discount INTEGER NOT NULL,
	pic_url TEXT NOT NULL UNIQUE
);

CREATE TABLE users(
	id_ INTEGER PRIMARY KEY AUTOINCREMENT,
	login TEXT NOT NULL UNIQUE,
	password TEXT NOT NULL UNIQUE,
	registration_date TEXT NOT NULL,
	birth_date TEXT NOT NULL
);

CREATE TABLE history(
	user_id INTEGER NOT NULL UNIQUE,
	item_id INTEGER NOT NULL UNIQUE,
	amount_ INTEGER NOT NULL,
	discount_ INTEGER NOT NULL,
	date TEXT NOT NULL,
	FOREIGN KEY (discount_) REFERENCES goods(discount),
	FOREIGN KEY (user_id) REFERENCES users(id_),
	FOREIGN KEY (item_id) REFERENCES goods(id)
);
